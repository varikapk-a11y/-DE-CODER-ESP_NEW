int hexStrToInt(String instring)
{
    byte len = instring.length();
    if  (len == 0) return 0;
    int result = 0;
    for (byte i = 0; i < 8; i++)    // —Ç–æ–ª—å–∫–æ –ø–µ—Ä–≤—ã–µ 8 —Ü—ã—Ñ–∞—Ä –≤–ª–µ–∑—É—Ç—å –≤ uint32
    {
        char ch = instring[i];
        if (ch == 0) break;
        result <<= 4;
        if (isdigit(ch))
        result = result | (ch - '0');
        else result = result | (ch - 'A' + 10);
    }
    return result;
}
